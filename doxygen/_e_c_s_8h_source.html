330  template<class T>
331  inline void ECS::VSG_SystemHelper<T>::record(vsg::RecordTraversal& rt) const
332  {
333  ROCKY_PROFILE_FUNCTION();
334 
335  // Get an optimized view of all this system's components:
336  auto view = registry.view<T>();
337 
338  using Entry = struct {
339  const T& component;
340  entt::entity entity;
341  };
342 
343  // Sort components into render sets by pipeline.
344  // If this system doesn't support multiple pipelines, just
345  // store them all together
346  std::vector<std::vector<Entry>> render_set(!pipelines.empty() ? pipelines.size() : 1);
347 
348  view.each([&](const entt::entity entity, const T& component)
349  {
350  // Is the component visible?
351  if (*component.active_ptr)
352  {
353  // Does it have a VSG node? If so, queue it up under the
354  // appropriate pipeline.
355  if (component.node)
356  {
357  auto& rs = !pipelines.empty() ? render_set[component.featureMask()] : render_set[0];
358  rs.emplace_back(Entry{ component, entity });
359  }
360 
361  // Otherwise, it's new and needs intialization, so queue it up for that.
362  else
363  {
364  entities_to_initialize.push_back(entity);
365  }
366  }
367  });
368 
369  // Time to record all visible components.
370  // For each pipeline:
371  for (int p = 0; p < render_set.size(); ++p)
372  {
373  if (!render_set[p].empty())
374  {
375  // Bind the Graphics Pipeline for this render set, if there is one:
376  if (!pipelines.empty())
377  {
378  pipelines[p].commands->accept(rt);
379  }
380 
381  // Them record each component.
382  // If the component has a transform apply it too.
383  for (auto& e : render_set[p])
384  {
385  auto* xform = registry.try_get<EntityTransform>(e.entity);
386  if (xform)
387  {
388  if (xform->node->push(rt))
389  {
390  e.component.node->accept(rt);
391  xform->node->pop(rt);
392  }
393  }
394  else
395  {
396  e.component.node->accept(rt);
397  }
398  }
399  }
400  }
401  }

403  template<class T>
404  inline void ECS::VSG_SystemHelper<T>::initializeComponents(Runtime& runtime)
405  {
406  ROCKY_PROFILE_FUNCTION();
407 
408  // Components with VSG elements need to create and compile those
409  // elements before we can render them. These get put on the
410  // initialization list by the record traversal.
411  if (!entities_to_initialize.empty())
412  {
413  NodeComponent::Params params;
414  params.readerWriterOptions = runtime.readerWriterOptions;
415 
416  for (auto& entity : entities_to_initialize)
417  {
418  auto& component = registry.get<T>(entity);
419  if (!component.node)
420  {
421  // if we're using pipelines, find the one matching this
422  // component's feature set:
423  if (pipelines.empty())
424  params.layout = { };
425  else
426  params.layout = pipelines[component.featureMask()].config->layout;
427 
428  // Tell the component to create its VSG node(s)
429  component.initializeNode(params);
430 
431  // TODO: Replace this will some error checking
432  ROCKY_SOFT_ASSERT(component.node);
433  }
434 
435  // compile the vulkan objects
436  if (component.node)
437  {
438  runtime.compile(component.node);
439  }
440  }
441 
442  // reset the list for the next frame
443  entities_to_initialize.clear();
444  }
445  }
