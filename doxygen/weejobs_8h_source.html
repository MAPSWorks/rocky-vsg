471  void dispatch(std::function<bool()>& delegate, const context& context)
472  {
473  // If we have a group semaphore, acquire it BEFORE queuing the job
474  if (context.group)
475  {
476  context.group->acquire();
477  }
478 
479  if (_targetConcurrency > 0)
480  {
481  std::unique_lock<std::mutex> lock(_queueMutex);
482  if (!_done)
483  {
484  _queue.emplace_back(job{ context, delegate });
485  _metrics.pending++;
486  _metrics.total++;
487  _block.notify_one();
488  }
489  }
490  else
491  {
492  // no threads? run synchronously.
493  delegate();
494 
495  if (context.group)
496  {
497  context.group->release();
498  }
499  }

513  void run()
514  {
515  while (!_done)
516  {
517  job next;
518  bool have_next = false;
519  {
520  std::unique_lock<std::mutex> lock(_queueMutex);
521 
522  _block.wait(lock, [this] {
523  return _queue.empty() == false || _done == true;
524  });
525 
526  if (!_queue.empty() && !_done)
527  {
528  // Find the highest priority item in the queue.
529  // Note: We could use std::partial_sort or std::nth_element,
530  // but benchmarking proves that a simple brute-force search
531  // is always the fastest.
532  // (Benchmark: https://stackoverflow.com/a/20365638/4218920)
533  // Also note: it is indeed possible for the results of
534  // priority() to change during the search. We don't care.
535  int index = -1;
536  float highest_priority = -FLT_MAX;
537  for (unsigned i = 0; i < _queue.size(); ++i)
538  {
539  float priority = _queue[i].ctx.priority != nullptr ?
540  _queue[i].ctx.priority() :
541  0.0f;
542 
543  if (index < 0 || priority > highest_priority)
544  {
545  index = i;
546  highest_priority = priority;
547  }
548  }
549  if (index < 0)
550  index = 0;
551 
552  next = std::move(_queue[index]);
553  have_next = true;
554 
555  // move the last element into the empty position:
556  if (index < _queue.size() - 1)
557  {
558  _queue[index] = std::move(_queue.back());
559  }
560 
561  // and remove the last element.
562  _queue.erase(_queue.end() - 1);
563  }
564  }
565 
566  if (have_next)
567  {
568  _metrics.running++;
569  _metrics.pending--;
570 
571  auto t0 = std::chrono::steady_clock::now();
572 
573  bool job_executed = next._delegate();
574 
575  auto duration = std::chrono::steady_clock::now() - t0;
576 
577  if (job_executed == false)
578  {
579  _metrics.canceled++;
580  }
581 
582  // release the group semaphore if necessary
583  if (next.ctx.group != nullptr)
584  {
585  next.ctx.group->release();
586  }
587 
588  _metrics.running--;
589  }
590 
591  // See if we no longer need this thread because the
592  // target concurrency has been reduced
593  std::lock_guard<std::mutex> lock(_quitMutex);
594 
595  if (_targetConcurrency < _metrics.concurrency)
596  {
597  _metrics.concurrency--;
598  break;
599  }
600  }
