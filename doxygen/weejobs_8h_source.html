597  inline bool _take_job(detail::job& output, bool lock)
598  {
599  if (lock)
600  {
601  std::lock_guard<std::mutex> lock(_queue_mutex);
602  return _take_job(output, false);
603  }
604  else if (!_done && _queue_size > 0)
605  {
606  auto ptr = _queue.end();
607  float highest_priority = -FLT_MAX;
608  for (auto iter = _queue.begin(); iter != _queue.end(); ++iter)
609  {
610  float priority = iter->ctx.priority != nullptr ?
611  iter->ctx.priority() :
612  0.0f;
613 
614  if (ptr == _queue.end() || priority > highest_priority)
615  {
616  ptr = iter;
617  highest_priority = priority;
618  }
619  }
620 
621  if (ptr == _queue.end())
622  ptr = _queue.begin();
623 
624  output = std::move(*ptr);
625  _queue.erase(ptr);
626  _queue_size--;
627  _metrics.pending--;
628  return true;
629  }
630  return false;

882  inline void jobpool::run()
883  {
884  while (!_done)
885  {
886  detail::job next;
887  bool have_next = false;
888  {
889  if (_can_steal_work && instance()._stealing_allowed)
890  {
891  {
892  std::unique_lock<std::mutex> lock(_queue_mutex);
893 
894  // work-stealing enabled: wait until any queue is non-empty
895  _block.wait(lock, [this]() { return get_metrics()->total_pending() > 0 || _done; });
896 
897  if (!_done && _queue_size > 0)
898  {
899  have_next = _take_job(next, false);
900  }
901  }
902 
903  if (!_done && !have_next)
904  {
905  have_next = detail::steal_job(this, next);
906  }
907  }
908  else
909  {
910  std::unique_lock<std::mutex> lock(_queue_mutex);
911 
912  // wait until just our local queue is non-empty
913  _block.wait(lock, [this] { return (_queue_size > 0) || _done; });
914 
915  if (!_done && _queue_size > 0)
916  {
917  have_next = _take_job(next, false);
918  }
919  }
920  }
921 
922  if (have_next)
923  {
924  _metrics.running++;
925 
926  auto t0 = std::chrono::steady_clock::now();
927 
928  bool job_executed = next._delegate();
929 
930  auto duration = std::chrono::steady_clock::now() - t0;
931 
932  if (job_executed == false)
933  {
934  _metrics.canceled++;
935  }
936 
937  // release the group semaphore if necessary
938  if (next.ctx.group != nullptr)
939  {
940  next.ctx.group->release();
941  }
942 
943  _metrics.running--;
944  }
945 
946  // See if we no longer need this thread because the
947  // target concurrency has been reduced
948  std::lock_guard<std::mutex> lock(_quit_mutex);
949 
950  if (_target_concurrency < _metrics.concurrency)
951  {
952  _metrics.concurrency--;
953  break;
954  }
955  }
956  }

1035  }
