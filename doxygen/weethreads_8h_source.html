528  void dispatch(std::function<bool()>& delegate, const context& context)
529  {
530  // If we have a group semaphore, acquire it BEFORE queuing the job
531  if (context.group)
532  {
533  context.group->acquire();
534  }
535 
536  if (_targetConcurrency > 0)
537  {
538  std::unique_lock<std::mutex> lock(_queueMutex);
539  if (!_done)
540  {
541  _queue.emplace_back(job{ context, delegate });
542  _metrics.pending++;
543  _metrics.total++;
544  _block.notify_one();
545  }
546  }
547  else
548  {
549  // no threads? run synchronously.
550  delegate();
551 
552  if (context.group)
553  {
554  context.group->release();
555  }
556  }

570  void run()
571  {
572  while (!_done)
573  {
574  job next;
575  bool have_next = false;
576  {
577  std::unique_lock<std::mutex> lock(_queueMutex);
578 
579  _block.wait(lock, [this] {
580  return _queue.empty() == false || _done == true;
581  });
582 
583  if (!_queue.empty() && !_done)
584  {
585  // Find the highest priority item in the queue.
586  // Note: We could use std::partial_sort or std::nth_element,
587  // but benchmarking proves that a simple brute-force search
588  // is always the fastest.
589  // (Benchmark: https://stackoverflow.com/a/20365638/4218920)
590  // Also note: it is indeed possible for the results of
591  // priority() to change during the search. We don't care.
592  int index = -1;
593  float highest_priority = -FLT_MAX;
594  for (unsigned i = 0; i < _queue.size(); ++i)
595  {
596  float priority = _queue[i].ctx.priority != nullptr ?
597  _queue[i].ctx.priority() :
598  0.0f;
599 
600  if (index < 0 || priority > highest_priority)
601  {
602  index = i;
603  highest_priority = priority;
604  }
605  }
606  if (index < 0)
607  index = 0;
608 
609  next = std::move(_queue[index]);
610  have_next = true;
611 
612  // move the last element into the empty position:
613  if (index < _queue.size() - 1)
614  {
615  _queue[index] = std::move(_queue.back());
616  }
617 
618  // and remove the last element.
619  _queue.erase(_queue.end() - 1);
620  }
621  }
622 
623  if (have_next)
624  {
625  _metrics.running++;
626  _metrics.pending--;
627 
628  auto t0 = std::chrono::steady_clock::now();
629 
630  bool job_executed = next._delegate();
631 
632  auto duration = std::chrono::steady_clock::now() - t0;
633 
634  if (job_executed == false)
635  {
636  _metrics.canceled++;
637  }
638 
639  // release the group semaphore if necessary
640  if (next.ctx.group != nullptr)
641  {
642  next.ctx.group->release();
643  }
644 
645  _metrics.running--;
646  }
647 
648  // See if we no longer need this thread because the
649  // target concurrency has been reduced
650  std::lock_guard<std::mutex> lock(_quitMutex);
651 
652  if (_targetConcurrency < _metrics.concurrency)
653  {
654  _metrics.concurrency--;
655  break;
656  }
657  }
